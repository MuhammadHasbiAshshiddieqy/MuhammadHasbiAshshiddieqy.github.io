4.1 ML Monitoring Fundamentals // kenapa APM tradisional tidak cukup untuk ML

Software konvensional gagal dengan cara yang deterministik: ada exception, ada stack trace, ada error log yang jelas. Model machine learning gagal dengan cara yang jauh lebih berbahaya — secara diam-diam. Model terus mengembalikan prediksi, latency normal, tidak ada error, tapi kualitas prediksi memburuk secara gradual selama berminggu-minggu. Pengguna mulai mendapatkan rekomendasi yang tidak relevan, fraud yang tidak terdeteksi, harga yang salah — sebelum tim menyadari ada masalah.

Inilah yang membuat monitoring ML secara fundamental berbeda dari Application Performance Monitoring (APM) tradisional. APM mengukur hal-hal seperti latency, throughput, error rate, dan resource utilization — semua metrik yang mengukur bagaimana sistem berjalan, bukan seberapa baik prediksi yang dihasilkan. Model bisa berjalan dengan sempurna secara infrastruktur tapi menghasilkan prediksi yang tidak berguna karena dunia telah berubah sejak model dilatih.

Metrik yang harus dimonitor di setiap layer

  • Data/Input metrics — distribusi setiap feature (mean, std, min, max, quantiles), null rate, schema violations, out-of-range values. Ini yang APM tradisional tidak pernah ukur.
  • Model output metrics — distribusi prediksi (apakah model tiba-tiba lebih banyak prediksi class 0?), prediction confidence/probability distribution, output range violations.
  • Business metrics — conversion rate, click-through rate, accuracy terhadap ground truth yang delayed (ground truth sering baru tersedia beberapa jam/hari kemudian). Ini adalah "truth" tertinggi.
  • System metrics — latency P50/P95/P99, throughput, memory, CPU, GPU utilization. APM tradisional sudah bagus di sini.
Ground truth latency adalah masalah terbesar di ML monitoringKamu bisa mengukur prediksi model secara real-time, tapi kamu tidak bisa mengukur akurasi secara real-time karena label yang benar sering baru tersedia nanti. Model churn prediction butuh 30 hari untuk tahu apakah prediksinya benar. Model fraud detection tahu dalam hitungan jam. Model rekomendasi tahu dalam hitungan menit dari click. Strategi monitoring harus dirancang sesuai ground truth latency masing-masing use case.

import numpy as np
import pandas as pd
from scipy import stats
from dataclasses import dataclass
from typing import Dict, List, Optional
import logging

logger = logging.getLogger(__name__)

@dataclass
class FeatureStats:
    """Statistik baseline untuk satu feature."""
    name: str
    mean: float
    std: float
    min_val: float
    max_val: float
    quantiles: Dict[str, float]  # {"p25": ..., "p50": ..., "p75": ..., "p95": ...}
    null_rate: float

class FeatureMonitor:
    """Monitor distribusi feature vs baseline training data."""

    def __init__(self, psi_threshold: float = 0.2, ks_alpha: float = 0.05):
        self.baseline_stats: Dict[str, FeatureStats] = {}
        self.psi_threshold = psi_threshold  # PSI > 0.2 = significant drift
        self.ks_alpha = ks_alpha            # KS test p-value threshold
        self.baseline_samples: Dict[str, np.ndarray] = {}

    def fit_baseline(self, df: pd.DataFrame, numerical_cols: List[str]) -> None:
        """Rekam statistik baseline dari training data."""
        for col in numerical_cols:
            values = df[col].dropna().values
            self.baseline_samples[col] = values
            self.baseline_stats[col] = FeatureStats(
                name=col,
                mean=float(values.mean()),
                std=float(values.std()),
                min_val=float(values.min()),
                max_val=float(values.max()),
                quantiles={
                    "p25": float(np.percentile(values, 25)),
                    "p50": float(np.percentile(values, 50)),
                    "p75": float(np.percentile(values, 75)),
                    "p95": float(np.percentile(values, 95)),
                },
                null_rate=float(df[col].isna().mean()),
            )
        logger.info(f"Baseline fitted for {len(numerical_cols)} features")

    def compute_psi(self, baseline: np.ndarray, current: np.ndarray,
                    n_bins: int = 10) -> float:
        """Population Stability Index: PSI < 0.1 = stabil, 0.1-0.2 = perlu investigasi, > 0.2 = drift signifikan."""
        bins = np.percentile(baseline, np.linspace(0, 100, n_bins + 1))
        bins[0] -= 1e-8; bins[-1] += 1e-8

        base_dist, _ = np.histogram(baseline, bins=bins, density=False)
        curr_dist, _ = np.histogram(current, bins=bins, density=False)

        # Normalize ke proporsi, hindari log(0)
        base_pct = (base_dist + 1e-8) / (len(baseline) + 1e-8)
        curr_pct = (curr_dist + 1e-8) / (len(current) + 1e-8)

        psi = np.sum((curr_pct - base_pct) * np.log(curr_pct / base_pct))
        return float(psi)

    def check_drift(self, current_df: pd.DataFrame) -> Dict[str, dict]:
        """Cek drift untuk semua feature yang di-monitor."""
        results = {}
        for col, baseline_vals in self.baseline_samples.items():
            if col not in current_df.columns:
                logger.warning(f"Feature {col!r} hilang dari current data!")
                continue
            current_vals = current_df[col].dropna().values
            psi = self.compute_psi(baseline_vals, current_vals)
            ks_stat, ks_pval = stats.ks_2samp(baseline_vals, current_vals)
            results[col] = {
                "psi": psi,
                "psi_status": "drift" if psi > self.psi_threshold else ("warning" if psi > 0.1 else "ok"),
                "ks_statistic": ks_stat,
                "ks_pvalue": ks_pval,
                "ks_drift": ks_pval < self.ks_alpha,
                "null_rate_current": float(current_df[col].isna().mean()),
            }
        return results

4.2 Evidently AI // open-source ML monitoring yang praktis

Evidently AI adalah library open-source Python yang menyederhanakan pembuatan data drift reports, model quality reports, dan data quality checks. Dibandingkan mengimplementasikan PSI atau KS-test sendiri, Evidently memberikan framework yang konsisten, visualisasi langsung, dan bisa di-embed ke monitoring pipeline. Ini bukan monitoring system lengkap — Evidently adalah library analisis yang menghasilkan laporan, yang kemudian kamu integrasikan ke stack monitoring yang lebih besar.

Konsep utama Evidently: kamu punya reference dataset (biasanya training data atau data dari periode "sehat" yang diketahui) dan current dataset (data production yang ingin diperiksa). Evidently membandingkan keduanya menggunakan berbagai statistical tests dan menghasilkan laporan yang bisa disajikan sebagai HTML, JSON, atau dikirim ke monitoring dashboard.

4.3 Prometheus & Grafana untuk ML // custom metrics & visualisasi real-time

Evidently bagus untuk batch drift analysis, tapi kamu juga butuh real-time metrics: berapa banyak prediksi per detik? Berapa latency P99 model endpoint? Berapa persen request yang mendapat prediksi dengan confidence rendah? Untuk kebutuhan ini, Prometheus dan Grafana adalah standar industri — Prometheus sebagai time-series metrics database, Grafana sebagai visualization layer.

Filosofi Prometheus adalah pull-based: Prometheus secara aktif "scrape" (menarik) metrics dari setiap service melalui HTTP endpoint /metrics. Service kamu menggunakan library prometheus_client untuk mendefinisikan dan mengekspos metrics, lalu Prometheus polling endpoint tersebut setiap N detik dan menyimpan data time-series. Grafana kemudian query Prometheus menggunakan bahasa PromQL untuk membuat dashboard real-time.

# Counter: hanya naik, tidak pernah turun. Cocok untuk request count, error count.
PREDICTION_COUNTER = Counter(
    "ml_predictions_total",
    "Total number of predictions made",
    labelnames=["model_name", "model_version", "prediction_class"],
)

REQUEST_ERRORS = Counter(
    "ml_prediction_errors_total",
    "Total number of prediction errors",
    labelnames=["model_name", "error_type"],
)

# Histogram: distribusi nilai dalam bucket. Ideal untuk latency.
# Bucket didefinisikan dalam detik: 1ms, 5ms, 10ms, 25ms, 50ms, 100ms, 250ms, 500ms, 1s
PREDICTION_LATENCY = Histogram(
    "ml_prediction_latency_seconds",
    "Prediction latency in seconds",
    labelnames=["model_name", "model_version"],
    buckets=[0.001, 0.005, 0.01, 0.025,
             0.05, 0.1, 0.25, 0.5, 1.0, 2.5],
)

# Gauge: bisa naik dan turun. Cocok untuk nilai current state.
MODEL_CONFIDENCE_GAUGE = Gauge(
    "ml_prediction_confidence_avg",
    "Rolling average prediction confidence (probability) — last 1000 predictions",
    labelnames=["model_name"],
)

DRIFT_SCORE_GAUGE = Gauge(
    "ml_feature_drift_score",
    "Current PSI drift score per feature",
    labelnames=["model_name", "feature_name"],
)

DATA_FRESHNESS_GAUGE = Gauge(
    "ml_training_data_age_days",
    "Days since model was last trained",
    labelnames=["model_name"],
)

# === Decorator untuk instrument fungsi prediksi ===
def monitor_prediction(model_name: str, model_version: str) -> Callable:
    def decorator(func: Callable) -> Callable:
        @wraps(func)
        def wrapper(*args, **kwargs) -> Any:
            start = time.perf_counter()
            try:
                result = func(*args, **kwargs)
                elapsed = time.perf_counter() - start

                PREDICTION_LATENCY.labels(
                    model_name=model_name,
                    model_version=model_version,
                ).observe(elapsed)

                # Asumsi result adalah (prediction_class, probability)
                pred_class, prob = result
                PREDICTION_COUNTER.labels(
                    model_name=model_name,
                    model_version=model_version,
                    prediction_class=str(pred_class),
                ).inc()

                return result
            except Exception as e:
                REQUEST_ERRORS.labels(
                    model_name=model_name,
                    error_type=type(e).__name__,
                ).inc()
                raise
        return wrapper
    return decorator

import schedule
import time
from evidently.report import Report
from evidently.metrics import ColumnDriftMetric

class DriftExporter:
    """Jalankan Evidently drift check secara terjadwal,
    ekspor hasilnya ke Prometheus sebagai Gauge metrics.
    """

    def __init__(self, model_name: str, feature_cols: list, monitor: FeatureMonitor):
        self.model_name = model_name
        self.feature_cols = feature_cols
        self.monitor = monitor

    def run_and_export(self, current_df) -> None:
        """Hitung drift dan update Prometheus Gauge."""
        drift_results = self.monitor.check_drift(current_df)
        for feature, metrics in drift_results.items():
            DRIFT_SCORE_GAUGE.labels(
                model_name=self.model_name,
                feature_name=feature,
            ).set(metrics["psi"])

            if metrics["psi_status"] == "drift":
                print(
                    f"[DRIFT ALERT] Feature '{feature}' PSI = {metrics['psi']:.3f}"
                    f" (threshold: 0.2) — consider retraining!"
                )

# === Schedule: jalankan setiap jam, bukan setiap request ===
# Drift detection mahal secara komputasi — batch, bukan per-request
exporter = DriftExporter("churn_model", feature_cols, monitor)
schedule.every(1).hour.do(exporter.run_and_export, current_df=get_hourly_batch())

while True:
    schedule.run_pending()
    time.sleep(60)

4.4 Alerting & Incident Response // dari alert ke tindakan dalam hitungan menit

Monitoring tanpa alerting adalah seperti smoke detector tanpa alarm — kamu tahu ada asap tapi tidak ada yang dibangunkan. Di sisi lain, alerting yang terlalu agresif menghasilkan alert fatigue: tim berhenti merespons karena terlalu banyak noise. Desain alerting yang baik adalah keahlian tersendiri, dan prinsip utamanya adalah: setiap alert yang sampai ke pager manusia harus actionable dan memerlukan respons segera. Jika tidak bisa direspons, itu bukan alert — itu logging.

Untuk ML systems, alerting lebih kompleks dari software biasa karena ada dua layer: alerting infrastruktur (latency, error rate) yang urgent dan relatif mudah di-threshold, dan alerting kualitas model (drift, accuracy degradation) yang lebih nuanced — sering kali threshold perlu disesuaikan dengan business context dan tidak ada angka "benar" yang universal.

SLO dan SLA untuk ML systems

  • SLO (Service Level Objective) — target internal: "model latency P99 < 200ms diukur per 30 hari rolling window". SLO lebih strict dari SLA karena merupakan warning sebelum SLA dilanggar. Tim platform bertanggung jawab ke SLO.
  • SLA (Service Level Agreement) — komitmen ke customer: "99.5% uptime per bulan, P99 latency < 500ms". Jika dilanggar ada konsekuensi finansial atau reputasi. SLA harus lebih longgar dari SLO.
  • Error Budget — jika SLO adalah 99.9% uptime/bulan, error budget adalah 0.1% = ~44 menit/bulan. Gunakan error budget untuk guide pengambilan keputusan: jika budget hampir habis, freeze deployment baru sampai akhir bulan.
  • ML-specific SLOs — selain latency dan uptime, pertimbangkan: "model accuracy tidak boleh turun > 5% dari baseline selama 7 hari rolling", "tidak lebih dari 3 critical drift alerts per bulan tanpa acknowledged retraining".
Alert fatigue adalah ancaman nyata untuk ML teamsTim yang menerima 50+ alert per hari akan berhenti merespons secara serius. Gejala: alert dibiarkan terlalu lama, "snooze everything" jadi budaya, incident yang mestinya P1 direspons seperti P3. Solusi: lakukan alert audit triwulanan, hapus alert yang tidak pernah actionable, pastikan setiap alert punya runbook yang jelas, dan investasikan dalam alert deduplication dan intelligent grouping di Alertmanager.

4.5 Retraining Triggers // kapan dan bagaimana retrain model secara otomatis

Retraining adalah "obat" untuk data drift dan concept drift — tapi seperti obat, timing dan dosisnya penting. Retrain terlalu jarang dan model akan terdegradasi secara diam-diam. Retrain terlalu sering dan kamu membuang compute, mungkin overfitting ke noise terbaru, dan berpotensi menghapus knowledge dari data lama yang masih relevan. Tantangan terbesar: bagaimana mengetahui kapan saat yang tepat untuk retrain?

Ada dua strategi utama: scheduled retraining (retrain setiap hari/minggu/bulan terlepas dari kondisi) dan drift-triggered retraining (retrain hanya ketika monitoring mendeteksi degradasi). Keduanya memiliki trade-off berbeda, dan banyak production system menggabungkan keduanya.

import mlflow
import mlflow.sklearn
import pandas as pd
from datetime import datetime, timedelta
from dataclasses import dataclass
from typing import Optional
import logging

logger = logging.getLogger(__name__)

@dataclass
class RetrainingConfig:
    model_name: str
    # Scheduled: retrain setiap N hari, bahkan tanpa drift
    max_age_days: int = 30
    # Drift-triggered: retrain jika PSI di atas threshold
    psi_threshold: float = 0.25
    # Akurasi: retrain jika akurasi turun lebih dari ini
    accuracy_drop_threshold: float = 0.05
    # Data minimum untuk retrain (jangan retrain dengan data terlalu sedikit)
    min_samples_for_retrain: int = 5000

class RetrainingDecider:
    """Logika untuk memutuskan kapan retrain diperlukan."""

    def __init__(self, config: RetrainingConfig):
        self.config = config

    def should_retrain(
        self,
        current_psi_scores: dict,          # {feature_name: psi_score}
        current_accuracy: Optional[float], # None jika ground truth belum tersedia
        baseline_accuracy: float,
        model_age_days: int,
        available_samples: int,
    ) -> tuple[bool, str]:
        """Returns (should_retrain, reason)."""

        if available_samples < self.config.min_samples_for_retrain:
            return False, f"Not enough data: {available_samples} < {self.config.min_samples_for_retrain}"

        # Trigger 1: Model terlalu tua
        if model_age_days >= self.config.max_age_days:
            return True, f"Scheduled: model age {model_age_days} days >= {self.config.max_age_days}"

        # Trigger 2: Feature drift
        drifted = {f: s for f, s in current_psi_scores.items() if s > self.config.psi_threshold}
        if drifted:
            top_drifted = max(drifted, key=lambda k: drifted[k])
            return True, f"Drift detected: {top_drifted} PSI={drifted[top_drifted]:.3f}"

        # Trigger 3: Accuracy degradation (hanya jika ground truth tersedia)
        if current_accuracy is not None:
            drop = baseline_accuracy - current_accuracy
            if drop > self.config.accuracy_drop_threshold:
                return True, f"Accuracy drop: {drop:.3f} > threshold {self.config.accuracy_drop_threshold}"

        return False, "No retraining needed"

def run_retraining_pipeline(
    config: RetrainingConfig,
    train_df: pd.DataFrame,
    val_df: pd.DataFrame,
) -> str:
    """Jalankan retraining dan register model baru ke MLflow Registry."""
    with mlflow.start_run(run_name=f"retrain-{datetime.utcnow():%Y%m%d-%H%M}"):
        mlflow.set_tags({
            "trigger": "drift",
            "model_name": config.model_name,
            "train_samples": len(train_df),
        })

        # ... train model ...
        model = train_model(train_df)
        metrics = evaluate_model(model, val_df)

        mlflow.sklearn.log_model(model, "model")
        mlflow.log_metrics(metrics)

        # Register ke Model Registry untuk enable staged rollout
        run_id = mlflow.active_run().info.run_id
        model_uri = f"runs:/{run_id}/model"
        mlflow.register_model(model_uri, config.model_name)
        logger.info(f"New model registered: {config.model_name}, run_id={run_id}")
        return run_id

Continual Learning dan Catastrophic Forgetting

Strategi di atas adalah full batch retraining — buang model lama, latih dari nol dengan data terbaru. Ini aman dan predictable, tapi ada alternatif: online learning atau continual learning, di mana model di-update secara inkremental dengan data baru tanpa melupakan data lama.

  • Catastrophic forgetting adalah masalah utama continual learning: ketika model di-update hanya dengan data baru, ia "melupakan" pola yang dipelajari dari data lama. Neural network sangat rentan terhadap ini — gradient update untuk task baru overwrite weight yang relevan untuk task lama.
  • Elastic Weight Consolidation (EWC) — menambah regularization yang melindungi weight "penting" dari data lama dari perubahan terlalu besar. Menggunakan Fisher Information Matrix untuk identifikasi weight yang kritis.
  • Replay buffer / Experience replay — saat training dengan data baru, mix-in sebagian data lama secara acak. Simple dan efektif untuk banyak kasus. Di-popularize dari deep reinforcement learning (DQN).
  • Untuk sistem production: kecuali ada kebutuhan spesifik untuk latency sangat rendah atau skenario streaming, batch retraining scheduled dengan windowed data (misalnya: 6 bulan terakhir) adalah strategi yang paling robust dan mudah di-debug.

from river import linear_model, preprocessing, metrics, stream
import pandas as pd
from collections import deque
import random

# === Online learning dengan River library ===
# River adalah library khusus untuk streaming/incremental ML

model = preprocessing.StandardScaler() | linear_model.LogisticRegression()
metric = metrics.Accuracy()
replay_buffer = deque(maxlen=1000)  # simpan 1000 sample lama untuk replay

def update_model_online(features: dict, label: int) -> float:
    """Update model dengan satu sample baru + experience replay.
    Returns current prediction accuracy.
    """
    # Prediksi dulu sebelum update (untuk evaluasi fairness)
    pred = model.predict_one(features)
    metric.update(label, pred)

    # Update dengan sample baru
    model.learn_one(features, label)

    # Experience replay: update juga dengan satu sample lama secara acak
    # Ini mengurangi catastrophic forgetting secara simple
    if len(replay_buffer) > 50:
        old_x, old_y = random.choice(replay_buffer)
        model.learn_one(old_x, old_y)

    # Simpan ke replay buffer
    replay_buffer.append((features, label))
    return metric.get()

# === Kapan TIDAK pakai online learning ===
# 1. Model Anda adalah neural network besar (catastrophic forgetting lebih severe)
# 2. Label tersedia dengan delay besar (tidak bisa update segera)
# 3. Butuh reproducibility audit trail (online learning harder to audit)
# 4. Distribution shift sangat tiba-tiba (model online bisa "mislearn" cepat)
#
# Gunakan online learning untuk:
# 1. Model sederhana (linear models, shallow trees)
# 2. High-throughput streaming data (click prediction, ad CTR)
# 3. Kasus di mana recency sangat penting (user behavior yang berubah cepat)

# === Windowed batch retraining — best of both worlds ===
def get_training_window(
    full_history_df: pd.DataFrame,
    window_days: int = 90,
    warmup_days: int = 30,   # data lama untuk mencegah forgetting
) -> pd.DataFrame:
    """Ambil window data untuk retraining: recent data + sample dari data lama.
    Strategi hybrid: batch retrain tapi dengan data yang ada temporal diversity.
    """
    cutoff = pd.Timestamp.now() - pd.Timedelta(days=window_days)
    recent = full_history_df[full_history_df["timestamp"] >= cutoff]

    # Sample data lebih tua untuk regularization — jaga stabilitas model
    old_data = full_history_df[full_history_df["timestamp"] < cutoff]
    warmup_cutoff = pd.Timestamp.now() - pd.Timedelta(days=window_days + warmup_days)
    warmup = old_data[old_data["timestamp"] >= warmup_cutoff].sample(
        n=min(len(warmup), 2000), replace=False
    )

    return pd.concat([recent, warmup]).reset_index(drop=True)

Gunakan shadow mode: model baru menerima semua request dan membuat prediksi, tapi prediksinya tidak digunakan untuk keputusan nyata — hanya di-log. Bandingkan distribusi output model lama vs baru selama 2–24 jam. Jika distribusi serupa dan tidak ada anomali, baru lakukan gradual rollout (5% → 20% → 50% → 100%) menggunakan feature flags atau A/B testing framework.

4.6 A/B Testing untuk ML // validasi model di production, bukan di lab

A/B testing untuk ML secara fundamental berbeda dari A/B testing untuk UI. Di UI, kamu mengubah warna tombol dan mengukur conversion — hasilnya terlihat dalam hitungan jam, sample size mudah dihitung, dan metrics deterministic. Di ML, kamu membandingkan dua prediktor probabilistik — metrics lebih noisy, efek lebih kecil, dan ground truth sering delayed (kamu baru tahu akurasi model churn 30 hari kemudian). Ini mengubah hampir seluruh metodologi statistik yang digunakan.

Konsep kunci yang harus dipahami: online metrics vs offline metrics. Offline metrics (accuracy, F1, AUC) diukur pada holdout dataset — ini yang kamu lihat di MLflow. Tapi yang benar-benar penting adalah online metrics (revenue, user retention, click-through rate) yang diukur langsung dari perilaku pengguna di production. Model dengan accuracy 94% bisa kalah dari model dengan accuracy 92% dalam online metrics — karena metrik bisnis yang sebenarnya tidak selalu berkorelasi sempurna dengan metrik ML.

Komponen A/B test yang rigorous untuk ML

  • Sample size calculation — pakai power analysis (bukan "kita test 2 minggu"). Butuh minimal detectable effect (MDE), significance level (α = 0.05), dan statistical power (1-β = 0.8). Tools: statsmodels.stats.power atau kalkulator online Evan Miller.
  • Randomization unit — user_id (bukan session_id atau request_id). Satu user harus selalu dapat treatment yang sama. Kalau tidak, kamu measurement-nya invalid.
  • Ramp-up strategy — mulai 5% traffic ke model baru, monitor 24 jam, naikkan ke 25%, monitor lagi, baru ke 50/50. Jangan langsung 50/50 — kalau model baru ada bug, 50% user terdampak.
  • Guardrail metrics — metrik yang tidak boleh degradasi: latency, error rate, cost per request. Model mungkin lebih akurat tapi 3× lebih lambat — tradeoff yang harus eksplisit.
  • Peeking problem — jangan cek hasil setiap hari dan stop saat p-value < 0.05. Itu inflates false positive rate drastis. Tentukan durasi test di awal, jalankan sampai selesai.
Pythonab_test_framework.py
import numpy as np
from scipy import stats
from dataclasses import dataclass
from enum import Enum

class TestStatus(Enum):
    RUNNING = "running"
    WINNER_A = "winner_a"      # model existing menang
    WINNER_B = "winner_b"      # model baru menang
    INCONCLUSIVE = "inconclusive"  # sample terlalu kecil

@dataclass
class ABTestResult:
    status: TestStatus
    p_value: float
    relative_lift: float        # e.g. 0.03 = model baru +3%
    ci_95: tuple[float, float]   # 95% confidence interval untuk lift
    sample_size_a: int
    sample_size_b: int

def analyze_ab_test(
    conversions_a: list[int],  # 0/1 per user di grup A (existing)
    conversions_b: list[int],  # 0/1 per user di grup B (new model)
    alpha: float = 0.05,
    min_lift: float = 0.01,    # minimal lift yang dianggap signifikan
) -> ABTestResult:
    n_a, n_b = len(conversions_a), len(conversions_b)
    rate_a = sum(conversions_a) / n_a
    rate_b = sum(conversions_b) / n_b
    lift = (rate_b - rate_a) / rate_a

    # Two-proportion z-test
    count_a, count_b = sum(conversions_a), sum(conversions_b)
    z_stat, p_value = stats.proportions_ztest(
        [count_b, count_a], [n_b, n_a],
        alternative='two-sided'
    )

    # Confidence interval untuk difference (Newey-West style)
    se = np.sqrt(rate_a * (1 - rate_a) / n_a + rate_b * (1 - rate_b) / n_b)
    ci_low = (rate_b - rate_a) - 1.96 * se
    ci_high = (rate_b - rate_a) + 1.96 * se

    if p_value < alpha and lift > min_lift:
        status = TestStatus.WINNER_B
    elif p_value < alpha and lift < -min_lift:
        status = TestStatus.WINNER_A
    else:
        status = TestStatus.INCONCLUSIVE

    return ABTestResult(
        status=status, p_value=p_value, relative_lift=lift,
        ci_95=(ci_low / rate_a, ci_high / rate_a),
        sample_size_a=n_a, sample_size_b=n_b,
    )

# === Contoh: simulasi A/B test ===
np.random.seed(42)
# Model existing: conversion 10%
control = np.random.binomial(1, 0.10, 5000)
# Model baru: conversion 10.5% (0.5% absolute lift)
treatment = np.random.binomial(1, 0.105, 5000)

result = analyze_ab_test(control, treatment)
print(f"Status: {result.status.value}")
print(f"P-value: {result.p_value:.4f}")
print(f"Lift: {result.relative_lift:.2%}")
print(f"95% CI: [{result.ci_95[0]:.2%}, {result.ci_95[1]:.2%}]")
P-value < 0.05 bukan berarti "model baru lebih baik"Statistical significance mengukur "apakah perbedaan ini mungkin terjadi karena random chance", bukan "apakah perbedaan ini penting secara bisnis". Model baru bisa signifikan secara statistik (p=0.01) dengan lift 0.1% — secara bisnis tidak meaningful. Sebaliknya, lift 5% bisa tidak signifikan (p=0.15) karena sample size terlalu kecil. Keduanya harus dipertimbangkan bersama.

4.7 Feedback Loops // saat prediksi model mengubah data masa depan

Salah satu failure mode paling berbahaya dalam ML production adalah feedback loop: model membuat prediksi → prediksi tersebut memengaruhi perilaku sistem/pengguna → data baru yang dihasilkan mencerminkan prediksi model sebelumnya → model di-retrain pada data ini → model menjadi semakin bias atau semakin jauh dari realitas. Ini adalah vicious cycle yang bisa berjalan tanpa terdeteksi selama berbulan-bulan.

Contoh klasik: model rekomendasi yang merekomendasikan konten populer → pengguna melihat konten populer → data training baru didominasi konten populer → model semakin bias ke konten populer → konten niche tidak pernah direkomendasikan → diversity kolaps. Atau model fraud detection yang terlalu agresif → memblokir transaksi legitimate → pengguna komplain → model belajar bahwa "transaksi dari pelanggan yang komplain" = OK → model menjadi terlalu permisif.

Vicious Cycle — Bagaimana Feedback Loop Mencemari Data Training
1. Model membuat prediksi
Rekomendasi, fraud score, ranking — dipakai sistem production
2. Prediksi memengaruhi user behavior
User klik rekomendasi, transaksi diblokir, ranking mengubah exposure
3. Data baru mencerminkan bias prediksi
Data training berikutnya hanya berisi "apa yang model pilih untuk ditampilkan"
4. Retrain → model makin bias
↺ Loop berulang, diversity kolaps, model makin jauh dari realitas

↑ Feedback loop adalah alasan #1 kenapa model yang "akurat" saat deployment bisa menjadi useless dalam 3-6 bulan tanpa disadari.

Strategi mitigasi feedback loop

  • Holdout baseline yang tidak terkontaminasi — Sisakan sekelompok pengguna yang TIDAK mendapat prediksi model (atau dapat prediksi random). Bandingkan behavior grup ini dengan grup yang kena model. Ini adalah cara paling akurat untuk mengukur true model impact.
  • Epsilon-greedy exploration — Model tidak selalu memilih prediksi optimalnya. Dengan probabilitas kecil (ε), pilih random. Exploration ini mencegah model stuck di local optimum dan memberikan data tentang "apa yang terjadi jika kita memilih opsi lain".
  • Counterfactual evaluation — Gunakan importance sampling atau inverse propensity scoring untuk mengestimasi "apa yang akan terjadi jika kita menggunakan model lain". Teknik ini berasal dari contextual bandit literature.
  • Monitor diversity metrics — Selain accuracy, monitor apakah distribusi output model berubah: apakah model makin sedikit merekomendasikan kategori tertentu? Apakah fraud score distribution bergeser? Diversity collapse sering mendahului accuracy drop.
Pythonfeedback_detector.py
import numpy as np
from collections import defaultdict

class FeedbackLoopDetector:
    """Deteksi apakah model output distribution mulai collapse (feedback loop)."""
    def __init__(self, window_size: int = 1000):
        self.window = window_size
        self.history = defaultdict(list)  # metric_name → [values]
        self.baseline_distribution = None

    def set_baseline(self, predictions: np.ndarray):
        """Rekam distribusi output baseline (saat model sehat)."""
        self.baseline_distribution = np.histogram(
            predictions, bins=20, density=True
        )[0]

    def record_batch(self, predictions: np.ndarray) -> dict:
        """Rekam satu batch prediksi, return diversity metrics."""
        current_dist = np.histogram(
            predictions, bins=20, density=True
        )[0]

        metrics = {
            "n_unique": len(np.unique(predictions)),
            "entropy": self._distribution_entropy(current_dist),
            "mode_frequency": float(current_dist.max()),
        }

        if self.baseline_distribution is not None:
            # Jensen-Shannon divergence: 0 = identik, 1 = totally berbeda
            metrics["js_divergence"] = self._js_divergence(
                self.baseline_distribution, current_dist
            )
            metrics["diversity_warning"] = metrics["js_divergence"] > 0.3

        for k, v in metrics.items():
            self.history[k].append(v)
            if len(self.history[k]) > self.window:
                self.history[k].pop(0)

        return metrics

    def _distribution_entropy(self, dist: np.ndarray) -> float:
        dist = dist[dist > 0]
        return float(-np.sum(dist * np.log(dist)))

    def _js_divergence(self, p: np.ndarray, q: np.ndarray) -> float:
        m = (p + q) / 2
        return float((np.sum(p * np.log(p / m + 1e-10)) +
                        np.sum(q * np.log(q / m + 1e-10))) / 2)
Feedback loop berbeda dari data driftData drift terjadi ketika distribusi input berubah karena faktor eksternal (musim, tren pasar, perubahan user behavior). Feedback loop terjadi karena MODEL SENDIRI yang menyebabkan perubahan pada data masa depan. Solusi untuk drift adalah retraining. Solusi untuk feedback loop adalah exploration strategy + holdout groups — retraining saja justru memperburuk masalah.
